public class Point {

   private double y;

  

   public Point() {

   }

      super();

      this.x = x;

      this.y = y;

   }

   public int getX() {

      return x;

   }

   public void setX(int x) {

      this.x = x;

   }

   public double getY() {

      return y;

   }

   public void setY(double y) {

      this.y = y;

   }

   public String toString() {

      return "("+x+", "+y+")";

   }
